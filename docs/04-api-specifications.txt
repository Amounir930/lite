1. معايير API العامة
═══════════════════════════════════════════════════════════════════════════════

▌1.1 Base URL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
التطوير:  http://localhost:3000/api/v1
الإنتاج:   https://api.whatsapp-crm.com/api/v1

▌1.2 Headers المطلوبة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Content-Type: application/json
Accept: application/json
Authorization: Bearer {jwt_token}  (أو Cookie: whatsapp-crm-session)
X-API-Key: {api_key}               (للتكامل الخارجي)
X-Correlation-ID: {uuid}           (لتتبع الطلبات)

▌1.3 تنسيق الـ Response
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
جميع الـ Responses تتبع التنسيق الموحد:

// ✅ نجاح
{
  "success": true,
  "data": { ... },
  "meta": {
    "page": 1,
    "limit": 20,
    "total": 150,
    "totalPages": 8,
    "hasNext": true,
    "hasPrev": false
  }
}

// ❌ خطأ
{
  "success": false,
  "error": {
    "code": "ERROR_CODE",
    "message": "Human readable error message",
    "details": [
      { "field": "email", "message": "Invalid email format" }
    ],
    "requestId": "uuid-for-debugging"
  }
}

▌1.4 HTTP Status Codes
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
200 OK              → طلب ناجح
201 Created         → تم الإنشاء بنجاح
204 No Content      → تم الحذف بنجاح (بدون body)
400 Bad Request     → بيانات غير صحيحة
401 Unauthorized    → غير مصرح (تسجيل الدخول مطلوب)
403 Forbidden       → ممنوع (صلاحيات غير كافية)
404 Not Found       → المورد غير موجود
409 Conflict        → تضارب في البيانات
422 Unprocessable   → Validation Error
429 Too Many Requests→ تجاوز الحد المسموح
500 Internal Error  → خطأ في الخادم

13. Pagination & Filtering
═══════════════════════════════════════════════════════════════════════════════

▌13.1 Pagination
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
المعلمات:
  page: integer (default: 1)
  limit: integer (default: 20, max: 100)

الاستجابة:
  meta: {
    page: 1,
    limit: 20,
    total: 150,
    totalPages: 8,
    hasNext: true,
    hasPrev: false
  }

▌13.2 Cursor-based Pagination (للسجلات الكبيرة)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
المعلمات:
  cursor: string (timestamp or id)
  limit: integer

الاستجابة:
  {
    "data": [ ... ],
    "hasMore": true,
    "nextCursor": "2026-05-05T10:00:00Z"
  }

▌13.3 Filtering
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
المعاملات:
  search: string (بحث عام)
  status: string (فلترة بحالة)
  dateFrom: string (YYYY-MM-DD)
  dateTo: string (YYYY-MM-DD)
  sortBy: string (عمود الترتيب)
  sortOrder: string (asc|desc)

▌13.4 Advanced Filtering
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
GET /api/v1/contacts?filter[status]=active&filter[tagIds][]=uuid1&filter[tagIds][]=uuid2

أو باستخدام JSON:
GET /api/v1/contacts?filter={"status":"active","tagIds":["uuid1","uuid2"]}
