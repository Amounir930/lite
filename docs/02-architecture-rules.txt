1. مبادئ التصميم المعماري
═══════════════════════════════════════════════════════════════════════════════

▌1.1 مبدأ Separation of Concerns (SoC)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ ممنوع: كتابة منطق الأعمال (Business Logic) داخل المكونات
✅ إلزامي: فصل المنطق إلى Services → Repositories → Models

مثال صحيح:
  Component → calls → Service → calls → Repository → Database
  
مثال خاطئ:
  Component → fetch() مباشرة إلى Database

▌1.2 مبدأ DRY (Don't Repeat Yourself)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ ممنوع: تكرار أكثر من 3 أسطر كود في موقعين مختلفين
✅ إلزامي: استخراج الكود المكرر إلى Utility / Hook / Component

▌1.3 مبدأ KISS (Keep It Simple, Stupid)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ ممنوع: حل معقد لمسألة بسيطة
✅ إلزامي: أبسط حل يحقق المتطلبات

قاعدة: إذا احتاج الكود لتعليق (comment) لتوضيح ما يفعله، فهو معقد

▌1.4 مبدأ YAGNI (You Ain't Gonna Need It)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ ممنوع: بناء ميزات "قد نحتاجها مستقبلاً"
✅ إلزامي: بناء فقط ما هو مطلوب في Sprint الحالي

▌1.5 مبدأ Composition over Inheritance
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ ممنوع: استخدام Class Inheritance في React
✅ إلزامي: استخدام Composition (HOC, Render Props, Custom Hooks)

▌2.2 قواعد تسمية الملفات
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
النوع              القاعدة                     مثال
─────────────────────────────────────────────────────────────────────────────
المجلدات           kebab-case                 flow-builder/
ملفات TS/TSX       kebab-case                 use-auth.ts
المكونات           PascalCase                 FlowBuilder.tsx
الأنواع            PascalCase + Type          UserType.ts
الواجهات           PascalCase + I             IUserService.ts
الثوابت            UPPER_SNAKE_CASE           MAX_RETRY_COUNT
الدوال             camelCase                  handleSubmit
الHooks            camelCase + use-             useAuth
الServices          PascalCase + Service       AuthService

▌2.3 قواعد تصدير الملفات (Exports)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ ممنوع: export default للمكونات (يسبب مشاكل في Tree Shaking)
✅ إلزامي: Named Exports فقط

// ✅ صحيح
export function Button() { ... }
export const Button = () => { ... }

// ❌ خاطئ
export default function Button() { ... }

❌ ممنوع: Barrel files (index.ts) مع re-export للجميع
✅ إلزامي: تصدير فقط ما هو مطلوب

// ✅ صحيح - index.ts
export { Button } from './button'
export { Input } from './input'

// ❌ خاطئ
export * from './button'
export * from './input'

3. قواعد SOLID المطبقة
═══════════════════════════════════════════════════════════════════════════════

▌3.1 Single Responsibility Principle (SRP)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
قاعدة: كل ملف/دالة/مكون مسؤولية واحدة فقط

❌ ممنوع: دالة تقوم بـ 3 مهام مختلفة
✅ إلزامي: تقسيم إلى 3 دوال منفصلة

// ❌ خاطئ
async function createContactAndSendMessage(data) {
  // 1. التحقق من البيانات
  // 2. حفظ في قاعدة البيانات
  // 3. إرسال رسالة ترحيب
}

// ✅ صحيح
async function validateContactData(data) { ... }
async function saveContact(data) { ... }
async function sendWelcomeMessage(contactId) { ... }

▌3.2 Open/Closed Principle (OCP)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
قاعدة: مفتوح للتوسع، مغلق للتعديل

❌ ممنوع: تعديل كود موجود لإضافة ميزة جديدة
✅ إلزامي: إضافة كود جديد عبر Extension Points

// ✅ صحيح - استخدام Strategy Pattern
interface MessageProvider {
  send(to: string, content: string): Promise<void>
}

class EvolutionProvider implements MessageProvider { ... }
class MetaProvider implements MessageProvider { ... }
class MockProvider implements MessageProvider { ... } // للاختبار

▌3.3 Liskov Substitution Principle (LSP)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
قاعدة: الفئة الفرعية يجب أن تكون قابلة للاستبدال بالفئة الأم

❌ ممنوع: Override لدالة ترمي Error غير متوقعة
✅ إلزامي: الحفاظ على العقد (Contract) للفئة الأم

▌3.4 Interface Segregation Principle (ISP)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
قاعدة: تقسيم الواجهات الكبيرة إلى واجهات صغيرة متخصصة

// ❌ خاطئ - واجهة ضخمة
interface IUserService {
  createUser()
  updateUser()
  deleteUser()
  sendEmail()
  generateReport()
  exportData()
}

// ✅ صحيح - واجهات منفصلة
interface IUserCRUDService { createUser(); updateUser(); deleteUser(); }
interface IEmailService { sendEmail(); }
interface IReportService { generateReport(); exportData(); }

▌3.5 Dependency Inversion Principle (DIP)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
قاعدة: الاعتماد على التجريدات، لا على التفاصيل

❌ ممنوع: استيراد Implementation مباشرة
✅ إلزامي: استيراد Interface/Abstraction

// ❌ خاطئ
import { PostgreSQLContactRepository } from './db/postgresql'
const repo = new PostgreSQLContactRepository()

// ✅ صحيح
import { IContactRepository } from './interfaces'
const repo: IContactRepository = container.resolve('contactRepository')

4. قواعد Clean Architecture
═══════════════════════════════════════════════════════════════════════════════

▌4.1 قواعد الاعتمادات (Dependency Rules)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
القاعدة الذهبية: الاعتمادات تتجه للداخل فقط

        ┌─────────────────────────┐
        │   Presentation Layer    │  ← Components, Pages
        │   (app/, components/)   │
        ├─────────────────────────┤
        │   Application Layer     │  ← Hooks, Services
        │   (hooks/, services/)   │
        ├─────────────────────────┤

// ✅ صحيح - lib/di/container.ts
class ServiceContainer {
  private services = new Map()
  
  register<T>(key: string, factory: () => T): void {
    this.services.set(key, { factory, instance: null })
  }
  
  resolve<T>(key: string): T {
    const service = this.services.get(key)
    if (!service.instance) {
      service.instance = service.factory()
    }
    return service.instance
  }
}

export const container = new ServiceContainer()

// تسجيل الخدمات

container.register('contactService', () => new ContactService(
  container.resolve('contactRepository')

export class AppError extends Error {
  constructor(
    public code: string,
    message: string,
    public statusCode: number = 500,
    public details?: Record<string, unknown>
  ) {
    super(message)
  }
}

export function handleApiError(error: unknown): Response {
  if (error instanceof AppError) {
    return Response.json(
      { success: false, error: { code: error.code, message: error.message, details: error.details } },
      { status: error.statusCode }
    )
  }
  
  // Log to monitoring service (Sentry)
  logger.error('Unhandled error', error)
  
  return Response.json(
    { success: false, error: { code: 'INTERNAL_ERROR', message: 'An unexpected error occurred' } },
    { status: 500 }
  )
}

// ✅ صحيح - middleware/correlation.ts
import { NextRequest, NextResponse } from 'next/server'
import { randomUUID } from 'crypto'

export function correlationMiddleware(request: NextRequest) {
  const correlationId = request.headers.get('x-correlation-id') || randomUUID()
  request.headers.set('x-correlation-id', correlationId)
  
  const response = NextResponse.next()
  response.headers.set('x-correlation-id', correlationId)
  return response
}

export async function getSession(request: Request) {

}

❌ ممنوع: التحقق من الصلاحيات في Client-side فقط
✅ إلزامي: التحقق على Server-side في كل API Route

// ✅ صحيح - middleware/rbac.ts
export function requirePermission(permission: string) {
  return async (request: NextRequest) => {
    const session = await getSession(request)
    if (!session?.user) {
      return Response.json({ error: 'Unauthorized' }, { status: 401 })
    }
    
    const hasPermission = await checkPermission(session.user.id, permission)
    if (!hasPermission) {
      return Response.json({ error: 'Forbidden' }, { status: 403 })
    }
    
    return NextResponse.next()
  }
}

▌12.5 قواعد Rate Limiting
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ ممنوع: API Routes بدون Rate Limiting
✅ إلزامي: Rate Limiting على جميع API Routes

// ✅ صحيح - lib/rate-limit.ts
import { Ratelimit } from '@upstash/ratelimit'
import { Redis } from '@upstash/redis'

const ratelimit = new Ratelimit({
  redis: Redis.fromEnv(),
  limiter: Ratelimit.slidingWindow(100, '15 m'),
  analytics: true,
})

export async function rateLimit(request: NextRequest) {
  const ip = request.ip ?? '127.0.0.1'
  const { success, limit, remaining, reset } = await ratelimit.limit(ip)
  
  if (!success) {
    return Response.json(
      { error: 'Rate limit exceeded' },
      { status: 429, headers: { 'X-RateLimit-Reset': reset.toString() } }
    )
  }
  
  const response = NextResponse.next()
  response.headers.set('X-RateLimit-Limit', limit.toString())
  response.headers.set('X-RateLimit-Remaining', remaining.toString())
  return response
}
