▌1.1 هرمية الاختبار (Testing Pyramid)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

        /
       / \     E2E Tests (Playwright)          5%  ← أعلى التكلفة
      /   \    ──────────────────────────────────────
     /     \   Integration Tests (Vitest + MSW)  15%  ← تكلفة متوسطة
    /       \  ──────────────────────────────────────
   /_________\ Unit Tests (Vitest + RTL)        80%  ← أقل التكلفة

▌1.2 مبادئ الاختبار
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Shift-Left Testing: الاختبار يبدأ من مرحلة التصميم
• Automated First: تفضيل الاختبار الآلي على اليدوي
• Independent Tests: كل اختبار مستقل عن الآخر
• Deterministic: نفس المدخلات = نفس المخرجات دائماً
• Fast Feedback: اختبارات الوحدة < 100ms لكل اختبار
• Isolated: لا يوجد Side Effects بين الاختبارات

// tests/unit/services/contact-service.test.ts
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { ContactService } from '@/services/contact-service'
import { IContactRepository } from '@/types/contact'

// Arrange - ترتيب البيانات
// Act - تنفيذ العملية
// Assert - التحقق من النتيجة

describe('ContactService', () => {
  let mockRepository: IContactRepository
  let service: ContactService

  beforeEach(() => {
    mockRepository = {
      create: vi.fn(),
      findById: vi.fn(),
      findAll: vi.fn(),
      update: vi.fn(),
      delete: vi.fn(),
    }
    service = new ContactService(mockRepository)
  })

  describe('create', () => {
    it('should create a new contact with valid data', async () => {
      // Arrange
      const data = {
        name: 'Ahmed Mohammed',
        phone: '+966501234567',
        email: 'ahmed@example.com',
      }
      const expectedContact = { id: 'uuid-1', ...data }
      mockRepository.create.mockResolvedValue(expectedContact)

      // Act
      const result = await service.create(data)

      // Assert
      expect(result).toEqual(expectedContact)
      expect(mockRepository.create).toHaveBeenCalledWith(data)
      expect(mockRepository.create).toHaveBeenCalledTimes(1)
    })

    it('should throw ValidationError when phone is invalid', async () => {
      // Arrange
      const data = { name: 'Test', phone: 'invalid' }

      // Act & Assert
      await expect(service.create(data)).rejects.toThrow('Invalid phone number')
    })

    it('should throw ConflictError when phone already exists', async () => {
      // Arrange
      const data = { name: 'Test', phone: '+966501234567' }
      mockRepository.findByPhone = vi.fn().mockResolvedValue({ id: 'existing' })

      // Act & Assert
      await expect(service.create(data)).rejects.toThrow('Phone already exists')
    })
  })

  describe('findAll', () => {
    it('should return paginated contacts', async () => {
      // Arrange
      const filters = { page: 1, limit: 20, status: 'active' }
      const expected = {
        contacts: [{ id: '1', name: 'Contact 1' }],
        total: 100,
      }
      mockRepository.findAll.mockResolvedValue(expected)

      // Act
      const result = await service.findAll(filters)

      // Assert
      expect(result.contacts).toHaveLength(1)
      expect(result.total).toBe(100)
    })
  })
})

▌2.4 اختبارات الـ Hooks
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// tests/unit/hooks/use-contacts.test.ts
import { describe, it, expect, vi } from 'vitest'
import { renderHook, waitFor } from '@testing-library/react'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { useContacts } from '@/hooks/use-contacts'

const createWrapper = () => {
  const queryClient = new QueryClient({
    defaultOptions: { queries: { retry: false } },
  })
  return ({ children }: { children: React.ReactNode }) => (
    <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>
  )
}

describe('useContacts', () => {
  it('should fetch contacts successfully', async () => {
    // Arrange
    const mockContacts = [{ id: '1', name: 'Contact 1' }]
    vi.mocked(contactService.getAll).mockResolvedValue({
      contacts: mockContacts,
      total: 1,
    })

    // Act
    const { result } = renderHook(() => useContacts({ page: 1 }), {
      wrapper: createWrapper(),
    })

    // Assert
    await waitFor(() => expect(result.current.isLoading).toBe(false))
    expect(result.current.contacts).toEqual(mockContacts)
    expect(result.current.error).toBeNull()
  })

  it('should handle error state', async () => {
    // Arrange
    vi.mocked(contactService.getAll).mockRejectedValue(new Error('Network error'))

    // Act
    const { result } = renderHook(() => useContacts({ page: 1 }), {
      wrapper: createWrapper(),
    })

    // Assert
    await waitFor(() => expect(result.current.isError).toBe(true))
    expect(result.current.error?.message).toBe('Network error')
  })
})

▌3.2 اختبارات API Routes
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// tests/integration/api/contacts.test.ts
import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { createTestDatabase, teardownTestDatabase } from '@/tests/helpers/db'
import { createTestUser, getAuthToken } from '@/tests/helpers/auth'

describe('Contacts API', () => {
  let db: ReturnType<typeof createTestDatabase>
  let authToken: string
  let testUser: { id: string; tenantId: string }

  beforeAll(async () => {
    db = createTestDatabase()
    testUser = await createTestUser(db)
    authToken = await getAuthToken(testUser)
  })

  afterAll(async () => {
    await teardownTestDatabase(db)
  })

  describe('POST /api/contacts', () => {
    it('should create contact with valid data', async () => {
      // Arrange
      const data = {
        name: 'New Contact',
        phone: '+966509876543',
        email: 'new@example.com',
      }

      // Act
      const response = await fetch('http://localhost:3000/api/contacts', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${authToken}`,
        },
        body: JSON.stringify(data),
      })

      // Assert
      expect(response.status).toBe(201)
      const result = await response.json()
      expect(result.success).toBe(true)
      expect(result.data.contact.name).toBe(data.name)
      expect(result.data.contact.phone).toBe(data.phone)
    })

    it('should return 400 for invalid data', async () => {
      // Arrange
      const data = { name: '', phone: 'invalid' }

      // Act
      const response = await fetch('http://localhost:3000/api/contacts', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${authToken}`,
        },
        body: JSON.stringify(data),
      })

      // Assert
      expect(response.status).toBe(400)
      const result = await response.json()
      expect(result.success).toBe(false)
      expect(result.error.code).toBe('VALID_001')
    })

    it('should return 401 without auth token', async () => {
      // Act
      const response = await fetch('http://localhost:3000/api/contacts', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name: 'Test', phone: '+966501234567' }),
      })

      // Assert
      expect(response.status).toBe(401)
    })
  })

  describe('GET /api/contacts', () => {
    it('should return paginated contacts', async () => {
      // Act
      const response = await fetch(
        'http://localhost:3000/api/contacts?page=1&limit=10',
        {
          headers: { Authorization: `Bearer ${authToken}` },
        }
      )

      // Assert
      expect(response.status).toBe(200)
      const result = await response.json()
      expect(result.success).toBe(true)
      expect(Array.isArray(result.data.contacts)).toBe(true)
      expect(result.meta).toHaveProperty('page')
      expect(result.meta).toHaveProperty('total')
    })
  })
})

      to: '+966501234567',
      content: 'Test message',
      sessionId: 'session-1',

    { duration: '2m', target: 200 },   // Ramp up
    { duration: '5m', target: 200 },   // Steady state
    { duration: '2m', target: 0 },     // Ramp down
  ],
  thresholds: {
    http_req_duration: ['p(95)<200'],
    http_req_failed: ['rate<0.01'],
  },
}

export default function () {
  const response = http.get('https://api.whatsapp-crm.com/api/v1/contacts', {
    headers: {
      Authorization: `Bearer ${__ENV.AUTH_TOKEN}`,
    },
  })

  check(response, {
    'status is 200': (r) => r.status === 200,
    'response time < 200ms': (r) => r.timings.duration < 200,
    'has contacts array': (r) => {
      const body = JSON.parse(r.body)
      return Array.isArray(body.data.contacts)
    },
  })

  sleep(1)
}

// tests/e2e/responsive/contact-page.spec.ts
import { test, expect } from '@playwright/test'

test.describe('Responsive Design', () => {
  const viewports = [
    { name: 'mobile', width: 375, height: 667 },
    { name: 'tablet', width: 768, height: 1024 },
    { name: 'desktop', width: 1920, height: 1080 },
  ]

  for (const viewport of viewports) {
    test(`contact page on ${viewport.name}`, async ({ page }) => {
      await page.setViewportSize({ width: viewport.width, height: viewport.height })
      await page.goto('/contacts')
      
      // Verify no horizontal scroll
      const body = await page.locator('body').boundingBox()
      expect(body?.width).toBeLessThanOrEqual(viewport.width)
      
      // Verify sidebar behavior
      if (viewport.name === 'mobile') {
        await expect(page.locator('[data-testid="sidebar"]')).toBeHidden()
      } else {
        await expect(page.locator('[data-testid="sidebar"]')).toBeVisible()
      }
    })
  }
})

LEFT JOIN tags t ON ct.tag_id = t.id

ORDER BY c.created_at DESC
LIMIT 20;
