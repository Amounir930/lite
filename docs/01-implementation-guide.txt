4. قواعد الهندسة المعمارية
═══════════════════════════════════════════════════════════════════════════════

▌4.1 قواعد SOLID المعمول بها
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
S - Single Responsibility: كل مكون/دالة مسؤولية واحدة فقط
O - Open/Closed: مفتوح للتوسع، مغلق للتعديل
L - Liskov Substitution: قابلية استبدال الفئات الفرعية
I - Interface Segregation: تقسيم الواجهات الكبيرة
D - Dependency Inversion: الاعتماد على التجريدات

▌4.2 قواعد Clean Architecture
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
طبقة العرض (Presentation)     → Components, Pages
طبقة التطبيق (Application)    → Hooks, Services, Validators
طبقة النطاق (Domain)          → Types, Interfaces, Business Logic
طبقة البنية التحتية (Infra)   → DB, API Clients, External Services

▌4.3 قواعد التسمية (Naming Conventions)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
المجلدات:       kebab-case (e.g., flow-builder)
الملفات:        kebab-case (e.g., use-auth.ts)
المكونات:       PascalCase (e.g., FlowBuilder.tsx)
الدوال:         camelCase (e.g., handleSubmit)
الثوابت:        UPPER_SNAKE_CASE (e.g., MAX_RETRY_COUNT)
الأنواع:        PascalCase مع لاحقة Type (e.g., UserType)
الواجهات:       PascalCase مع لاحقة I (e.g., IUserRepository)

▌4.4 قواعد قاعدة البيانات
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
- جميع الجداول: snake_case
- المفاتيح الأساسية: uuid (NOT serial)
- الطوابع الزمنية: created_at, updated_at (إلزامي)
- التسميات العربية: jsonb للتعدد اللغوي
- الفهارس: على كل عميل foreign key + أعمدة البحث
- Soft Delete: عمود deleted_at (اختياري حسب الجدول)

▌8.1 حماية البيانات
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• تشفير كلمات المرور: bcrypt مع salt rounds = 12
• تشفير الجلسات: Iron-Session مع secret 32+ حرف
• تشفير البيانات الحساسة: AES-256-GCM
• TLS 1.3: إلزامي لجميع الاتصالات
• GDPR Compliance: تصدير/حذف بيانات المستخدم

▌8.2 Rate Limiting Strategy
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
المستوى 1: IP-based    → 100 req/15min
المستوى 2: User-based  → 1000 req/15min (مميز)
المستوى 3: API Key     → حسب الخطة
المستوى 4: WebSocket   → 50 msg/min

▌8.3 Audit Logging
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• تسجيل جميع عمليات المصادقة
• تسجيل تغييرات البيانات الحساسة
• تسجيل الوصول إلى API
• الاحتفاظ: 90 يوم للسجلات العادية، 1 سنة للسجلات الحرجة
